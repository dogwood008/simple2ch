def self.fetch(url)
  req = Net::HTTP::Get.new(url.path)
  res = Net::HTTP.start(url.host, url.port) {|http|
    http.request(req)
  }
  res.body.force_encoding("cp932").encode!('utf-8', :undef => :replace)
end
